# swsd-mcp v2.1.0 — Tool UX & UI Expansion

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Reduce friction on existing tools so agents can hand them human-natural inputs (numbers, dates), and dramatically expand the widget surface so MCP-Apps-capable hosts get rich, usable UI for the most-used workflows.

**Architecture:** Three-layer change. (1) Foundation: a shared id-vs-number resolver utility, a sanitized-HTML helper, and a tool-result error handler in `src/ui/shared/host.ts`. (2) Tool layer: bulk-apply id_or_number to 11 tools; add 6 new tools (3 tasks-family, 3 problems-family); add date-natural-language aliases to two list tools. (3) UI layer: 3 new widgets (comment-thread, audit-timeline, catalog-item-form), enrichments to 2 existing widgets (incident-detail richer fields, incident-list overflow + sort indicator), and HTML-body rendering on solution-detail.

**Tech Stack:** TypeScript 5.7, Zod schemas, MCP SDK `^1.29.0`, `@modelcontextprotocol/ext-apps@^1.7.1`, Vite + vite-plugin-singlefile for widget bundles, Vitest for tests, **DOMPurify `^3.4.2`** for HTML sanitization (verified 2026-05-07: latest is 3.4.2 published 2026-04-30, no peer-deps).

**Source of truth:** `.research/v2.1-audit-findings.md` (3,300-word audit completed 2026-05-07; cited throughout this plan).

**Out of scope (deferred to v2.2):** Changes-management tools (`/changes.json`), Releases, Reports, Memos, asset listing tools, attachment upload, catalog-list and solution-search-results card-grid widgets, overdue-only filter. These are P2 in the audit; v2.1 is already large.

```typescript
import { InputError } from '../errors';

const ID_DIGIT_THRESHOLD = 7;

export interface SwsdClientForResolver {
  listIncidents: (params: { numbers?: number[]; per_page?: number }) => Promise<{
    items: Array<{ id: number; number: number }>;
    totalCount: number;
  }>;
  listSolutions?: (params: { numbers?: number[]; per_page?: number }) => Promise<{
    items: Array<{ id: number; number: number }>;
    totalCount: number;
  }>;
}

function assertPositiveInteger(value: number, label: string): void {
  if (!Number.isInteger(value)) {
    throw new InputError(label + ' must be an integer (got ' + value + ')');
  }
  if (value <= 0) {
    throw new InputError(label + ' must be a positive number (got ' + value + ')');
  }
}

function digitCount(value: number): number {
  return Math.abs(value).toString().length;
}

export async function resolveIncidentRef(
  input: number,
  client: SwsdClientForResolver,
): Promise<{ id: number }> {
  assertPositiveInteger(input, 'incident reference');
  if (digitCount(input) >= ID_DIGIT_THRESHOLD) {
    return { id: input };
  }
  const result = await client.listIncidents({ numbers: [input], per_page: 1 });
  if (result.items.length === 0) {
    throw new InputError(
      'incident with number ' + input + ' not found. ' +
      'If ' + input + ' is the internal id, ensure it has at least ' + ID_DIGIT_THRESHOLD + ' digits.'
    );
  }
  return { id: result.items[0].id };
}

export async function resolveSolutionRef(
  input: number,
  client: SwsdClientForResolver,
): Promise<{ id: number }> {
  assertPositiveInteger(input, 'solution reference');
  if (digitCount(input) >= ID_DIGIT_THRESHOLD) {
    return { id: input };
  }
  if (!client.listSolutions) {
    throw new InputError(
      'solution must be referenced by id (>=' + ID_DIGIT_THRESHOLD + ' digits). ' +
      'Got ' + input + '; use swsd_search_solutions to discover the id.'
    );
  }
  const result = await client.listSolutions({ numbers: [input], per_page: 1 });
  if (result.items.length === 0) {
    throw new InputError('solution with number ' + input + ' not found.');
  }
  return { id: result.items[0].id };
}
```

- [ ] **Step 3: Update `src/ui/shared/host.ts` mountApp signature**

```typescript
export interface MountAppOptions<T> {
  name: string;
  version: string;
  onResult: (structuredContent: T) => void;
  onError?: (error: { message: string; structuredContent?: unknown }) => void;
}

export async function mountApp<T>(opts: MountAppOptions<T>): Promise<App> {
  const app = new App({ name: opts.name, version: opts.version });

  app.addEventListener('toolresult', (params) => {
    if (params.isError) {
      const message = extractTextMessage(params.content) ??
        'The tool reported an error but did not provide a message.';
      if (opts.onError) {
        opts.onError({ message, structuredContent: params.structuredContent });
      } else {
        console.warn('[mountApp] tool-result error (no onError handler):', message);
      }
      return;
    }
    const sc = params.structuredContent;
    if (sc !== undefined) opts.onResult(sc as T);
  });

  app.addEventListener('hostcontextchanged', (params) => {
    applyHostContextStyling(params);
  });

  await app.connect();
  const ctx = app.getHostContext();
  if (ctx) applyHostContextStyling(ctx);
  return app;
}

function extractTextMessage(content: unknown): string | undefined {
  if (!Array.isArray(content)) return undefined;
  for (const part of content) {
    if (part && typeof part === 'object' && 'type' in part && part.type === 'text' && 'text' in part) {
      const text = (part as { text: unknown }).text;
      if (typeof text === 'string' && text.length > 0) return text;
    }
  }
  return undefined;
}
```

import DOMPurify from 'dompurify';

const ALLOWED_TAGS = [
  'p', 'br', 'div', 'span',
  'h1', 'h2', 'h3', 'h4', 'h5', 'h6',
  'strong', 'em', 'b', 'i', 'u', 'code', 'pre', 'blockquote', 'cite',
  'ul', 'ol', 'li',
  'a', 'img',
  'table', 'thead', 'tbody', 'tr', 'th', 'td',
  'hr',
];

const ALLOWED_ATTR = [
  'href', 'rel', 'target', 'title',
  'src', 'alt', 'width', 'height',
  'class', 'id',
  'colspan', 'rowspan',
];

const FORBID_TAGS = ['script', 'style', 'iframe', 'object', 'embed', 'form', 'input', 'textarea', 'button', 'link', 'meta'];
const FORBID_ATTR = ['style'];

let configured = false;
function configure(): void {
  if (configured) return;
  configured = true;
  DOMPurify.addHook('afterSanitizeAttributes', (node) => {
    if (node instanceof HTMLAnchorElement) {
      node.setAttribute('target', '_blank');
      node.setAttribute('rel', 'noopener noreferrer');
    }
  });
}

export function sanitizeHtml(html: string): string {
  configure();
  return DOMPurify.sanitize(html, {
    ALLOWED_TAGS,
    ALLOWED_ATTR,
    FORBID_TAGS,
    FORBID_ATTR,
    ALLOW_DATA_ATTR: false,
  });
}

```typescript
const ALIAS_RE = /^(\d+)(d|h|w)$/;

export function parseDateAlias(input: string): string | null {
  const m = ALIAS_RE.exec(input.trim());
  if (!m) return null;
  const value = parseInt(m[1], 10);
  const unit = m[2];
  if (!Number.isFinite(value) || value <= 0 || value > 365) return null;

  const now = new Date();
  const msPerHour = 3600 * 1000;
  const msPerDay = 24 * msPerHour;
  const ms = unit === 'h' ? msPerHour : unit === 'd' ? msPerDay : 7 * msPerDay;
  const target = new Date(now.getTime() - value * ms);
  return target.toISOString().slice(0, 10);
}

export function applyDateAlias<T extends { updated_within?: string; updated_from?: string }>(
  input: T,
): T {
  if (!input.updated_within) return input;
  const parsed = parseDateAlias(input.updated_within);
  if (!parsed) return input;
  if (input.updated_from) return input;
  const { updated_within, ...rest } = input;
  return { ...rest, updated_from: parsed } as T;
}
```

export function toTask(raw: unknown): Task {

export function toProblem(raw: unknown): Problem {

interface CatalogVariable {
  id: number;
  custom_field_id: number;
  name: string;
  kind: 'free_text' | 'drop_down_menu' | 'multi_select' | 'date' | 'user' | 'null';
  field_type: number;
  options?: string;
  required: '0' | '1';
  helptext?: string;
}

interface CatalogItem {
  id: number;
  name: string;
  description?: string;
  description_no_html?: string;
  variables: CatalogVariable[];
}

let appHandle: { callServerTool: (name: string, args: unknown) => Promise<unknown> } | null = null;

function renderField(v: CatalogVariable): string {
  const required = v.required === '1';
  const inputId = 'field-' + v.id;
  const baseAttrs = 'id="' + inputId + '" name="' + v.custom_field_id + '" ' + (required ? 'required' : '');
  let control: string;
  switch (v.kind) {
    case 'drop_down_menu': {
      const options = (v.options ?? '').split('\n').filter(Boolean);
      control = '<select ' + baseAttrs + '>' +
        '<option value="" ' + (required ? 'disabled' : '') + '>' + (required ? 'Select...' : '(not selected)') + '</option>' +
        options.map(o => '<option>' + escapeHtml(o) + '</option>').join('') +
      '</select>';
      break;
    }
    case 'multi_select': {
      const options = (v.options ?? '').split('\n').filter(Boolean);
      control = '<select ' + baseAttrs + ' multiple size="4">' +
        options.map(o => '<option>' + escapeHtml(o) + '</option>').join('') +
      '</select>';
      break;
    }
    case 'date':
      control = '<input type="date" ' + baseAttrs + ' />';
      break;
    case 'user':
      control = '<input type="email" placeholder="user@example.com" ' + baseAttrs + ' />';
      break;
    default:
      control = '<input type="text" ' + baseAttrs + ' />';
  }
  return '<div class="field">' +
    '<label for="' + inputId + '">' +
      escapeHtml(v.name) + (required ? ' <span class="required">*</span>' : '') +
    '</label>' +
    control +
    (v.helptext ? '<div class="help-text prose">' + sanitizeHtml(v.helptext) + '</div>' : '') +
  '</div>';
}

function render(item: CatalogItem): void {
  const root = document.getElementById('root');
  if (!root) return;
  root.innerHTML =
    '<article class="catalog-form">' +
      '<header>' +
        '<h1>' + escapeHtml(item.name) + '</h1>' +
        (item.description ? '<div class="catalog-description prose">' + sanitizeHtml(item.description) + '</div>' : '') +
      '</header>' +
      '<form id="request-form" novalidate>' +
        item.variables.map(renderField).join('') +
        '<div class="form-actions">' +
          '<button type="submit" class="primary">Submit request</button>' +
          '<span class="form-status" role="status" aria-live="polite"></span>' +
        '</div>' +
      '</form>' +
    '</article>';
  const form = document.getElementById('request-form') as HTMLFormElement;
  form.addEventListener('submit', (e) => onSubmit(e, item));
}

async function onSubmit(e: SubmitEvent, item: CatalogItem): Promise<void> {
  e.preventDefault();
  if (!appHandle) return;
  const form = e.target as HTMLFormElement;
  const status = form.querySelector('.form-status') as HTMLElement;
  status.textContent = 'Submitting...';

  const fd = new FormData(form);
  const variables: Array<{ custom_field_id: number; value: string }> = [];
  for (const v of item.variables) {
    const value = fd.get(String(v.custom_field_id));
    if (value !== null && value !== '') {
      variables.push({ custom_field_id: v.custom_field_id, value: String(value) });
    }
  }

  try {
    // Method name verified at .d.ts location: see Step 1 citation.
    await appHandle.callServerTool('swsd_create_service_request', {
      catalog_item_id: item.id,
      requester_email: '',
      variables,
    });
    status.textContent = 'Submitted successfully.';
  } catch (err) {
    status.textContent = 'Submit failed: ' + (err instanceof Error ? err.message : String(err));
  }
}

const app = await mountApp<CatalogItem>({
  name: 'swsd-catalog-item-form',
  version: '1.0.0',
  onResult: render,
  onError: ({ message }) => {
    const root = document.getElementById('root');
    if (root) root.innerHTML = '<div class="error-state">' + escapeHtml(message) + '</div>';
  },
});
appHandle = app as unknown as typeof appHandle;

import { sanitizeHtml } from '../shared/sanitizeHtml';
import { escapeHtml } from '../shared/escape';

function isOverdue(dueAt?: string): boolean {
  return !!dueAt && new Date(dueAt) < new Date();

function classFromState(state: string): string {
  return state.toLowerCase().replace(/[^a-z0-9]+/g, '-');
}

function renderDetail(d: IncidentDetail): string {
  return '<article class="incident">' +
    '<header class="incident-header">' +
      '<h1>#' + d.number + ' — ' + escapeHtml(d.name) + '</h1>' +
      '<div class="incident-badges">' +
        '<span class="badge badge-state badge-state-' + classFromState(d.state) + '">' + escapeHtml(d.state) + '</span>' +
        (d.priority ? '<span class="badge badge-priority">' + escapeHtml(d.priority) + '</span>' : '') +
        (isOverdue(d.due_at) ? '<span class="badge badge-overdue">Overdue</span>' : '') +
        (d.sla_violations && d.sla_violations.length > 0
          ? '<span class="badge badge-sla">' + d.sla_violations.length + ' SLA violation(s)</span>'
          : '') +
      '</div>' +
    '</header>' +
    '<dl class="incident-meta">' +
      '<dt>Assignee</dt>           <dd>' + escapeHtml(d.assignee?.name ?? '—') + '</dd>' +
      '<dt>Requester</dt>          <dd>' + escapeHtml(d.requester?.name ?? '—') + '</dd>' +
      '<dt>Category</dt>           <dd>' + escapeHtml(d.category ?? '—') + '</dd>' +
      '<dt>Created</dt>            <dd>' + escapeHtml(formatTime(d.created_at ?? '')) + '</dd>' +
      '<dt>Updated</dt>            <dd>' + escapeHtml(formatTime(d.updated_at ?? '')) + '</dd>' +
      (d.due_at ? '<dt>Due</dt><dd class="' + (isOverdue(d.due_at) ? 'overdue' : '') + '">' +

    '</dl>' +
    (d.description
      ? '<section class="incident-description"><h2>Description</h2><div class="prose">' +
        sanitizeHtml(d.description) + '</div></section>'
      : '') +
    (d.resolution
      ? '<section class="incident-resolution"><h2>Resolution' +
        (d.resolution_type ? ' (' + escapeHtml(d.resolution_type) + ')' : '') + '</h2>' +
        '<div class="prose">' + sanitizeHtml(d.resolution) + '</div></section>'
